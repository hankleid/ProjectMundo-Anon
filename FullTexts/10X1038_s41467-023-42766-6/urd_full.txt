دنیا بھر میں آزاد گھومنے والی گھریلو بلیوں کی خوراک کا عالمی تجزیہ اور جائزہ

خلاصہ: آزاد گھومنے والی بلیاں (Felis catus ) دنیا بھر میں پھیلی ہوئی جارحانہ گوشت خور ہیں جو حیاتیاتی تنوع پر نمایاں اثر ڈالتی ہیں۔ یہاں، بلیوں کے ممکنہ خطرے کا اندازہ لگانے کے لیے، ہم بلیوں کے ذریعے کھائی جانے والی انواع کا ایک جامع عالمی جائزہ تیار کرتے ہیں۔ ہم نے 2,084 انواع کی نشاندہی کی ہے جو بلیوں کے ذریعے کھائی جاتی ہیں، جن میں سے 347 (16.65%) تحفظ کے لحاظ سے تشویش کا شکار ہیں۔ جزائر میں بلیوں کے ذریعے کھائی جانے والی تحفظ کے لحاظ سے تشویش کا شکار انواع کی تعداد براعظموں کے مقابلے میں تین گنا زیادہ ہے۔ پرندے، رینگنے والے جانور، اور ممالیہ تقریباً 90% کھائی جانے والی انواع پر مشتمل ہیں، جبکہ کیڑے اور ایمفیبین کم پائے جاتے ہیں۔ بلیوں کی خوراک میں تقریباً 9% معلوم پرندے، 6% معلوم ممالیہ، اور 4% معلوم رینگنے والے جانور شامل ہیں۔ کھائی جانے والی 97% انواع بالغ جسمانی وزن میں <5 کلوگرام ہیں، حالانکہ بہت بڑی انواع بھی کھائی جاتی ہیں۔ انواع کے جمع ہونے کے منحنی خطوط غیر متناسب نہیں ہیں، جو ظاہر کرتے ہیں کہ ہمارے تخمینے محتاط ہیں۔ ہمارے نتائج ظاہر کرتے ہیں کہ بلیاں انتہائی عمومی شکاری ہیں، جو ان کے ماحولیاتی نظاموں پر اثرات کو سمجھنے اور انتظامی حل تیار کرنے کے لیے اہم ہے۔

چونکہ گھریلو بلیاں (Felis catus ) 9000 سال سے زیادہ پہلے پالتو بنائی گئی تھیں، انسانوں نے انہیں دنیا کے بیشتر حصے میں متعارف کرایا ہے ۔ آج، بلیاں تمام براعظموں میں آباد ہیں، سوائے انٹارکٹیکا کے، اور انہیں سیکڑوں جزیروں میں متعارف کرایا گیا ہے، جس کی وجہ سے وہ سیارے پر سب سے زیادہ وسیع پیمانے پر پھیلی ہوئی انواع میں شامل ہیں۔ اس عالمی تقسیم کی وجہ سے، بلیوں نے ان بہت سے ماحولیاتی نظاموں کو متاثر کیا ہے جن میں انہیں متعارف کرایا گیا ہے ۔ خاص طور پر، بلیاں انسانوں سمیت مختلف انواع میں نئی ​​بیماریاں پھیلاتی ہیں، مقامی فیلڈز اور دیگر درمیانی شکاریوں سے مقابلہ کرتی ہیں، جنگلی فیلڈز کی جینیاتی سالمیت کو خطرے میں ڈالتی ہیں، مقامی جانوروں کا شکار کرتی ہیں، اور بہت سی انواع کو معدومیت کی طرف دھکیل چکی ہیں, ,۔ نتیجتاً، آزاد گھومنے والی بلیاں (یعنی، وہ بلیاں جو باہر کے ماحول تک رسائی رکھتی ہیں، چاہے وہ مالک کی ہوں یا نہ ہوں) دنیا کی سب سے زیادہ مسئلہ پیدا کرنے والی حملہ آور انواع میں شامل ہیں ۔

ایک خصوصیت جس نے بلیوں کو کامیاب حملہ آور بننے کی اجازت دی ہے وہ ان کی عمومی خوراک ہے۔ بلیاں موقع پرست شکاری اور لازمی گوشت خور ہیں جو کھانے میں موجود پانی اور میٹابولک پانی پر مہینوں تک زندہ رہ سکتی ہیں,۔ مزید برآں، بلیوں نے صرف جانوروں کے بافتوں پر زندہ رہنے کے لیے ارتقاء کیا ہے اور ان کے پاس گوشت خوروں کے طور پر مخصوص غذائی موافقتیں ہیں۔ خاص طور پر، بلیوں میں امینو ایسڈ میٹابولزم کے انزائمز کو منظم کرنے کی محدود صلاحیت ہوتی ہے، اور پودوں کے مواد کو امینو ایسڈز اور وٹامنز میں تبدیل کرنے کی صلاحیت نہیں ہوتی ۔ لہذا، اگرچہ بلیاں پودوں کے مواد کو کھاتی ہیں، وہ اپنی توانائی کی ضروریات کو پورا کرنے کے لیے اعلی پروٹین والی خوراک پر انحصار کرتی ہیں۔ ان جسمانی ضروریات اور رویے کی خصوصیات کے نتیجے میں، بلیاں مختلف قسم کے جانوروں کا شکار کرنے اور ان کی لاشوں کو کھانے کے لیے جانی جاتی ہیں, ,۔

بلیوں کے لیے دنیا بھر میں غذائی تجزیے 100 سال سے زیادہ عرصے سے کیے جا رہے ہیں، جن میں سے بہت سے مطالعات سے پتہ چلتا ہے کہ یا تو پرندے یا چھوٹے ممالیہ غالب شکار کی اشیاء ہیں، اکثر ان ماحولیاتی نظاموں پر منحصر ہوتا ہے جن میں مطالعات کی گئی تھیں, ,۔ مطالعات کے درمیان ایسی غذائی اختلافات ممکنہ طور پر جزوی طور پر شکار کی دستیابی میں مقامات کے درمیان اختلافات کی عکاسی کرتے ہیں ۔ لہذا، اگرچہ وسیع پیمانے پر پھیلی ہوئی انواع عام طور پر بلیوں کی خوراک میں پائی جاتی ہیں [مثلاً، گھریلو چوہا (Mus musculus ), گھریلو چڑیا (Passer domesticus )], یہ زیادہ تر مطالعہ کے مقام اور شکار کی تقسیم اور کثرت کی عکاسی ہے، بجائے اس کے کہ خوراک کی ترجیح ۔ تاہم، حالیہ نتائج سے پتہ چلتا ہے کہ بلیاں مخصوص شکار کی انواع کا شکار جاری رکھتی ہیں، یہاں تک کہ جب یہ شکار کی انواع نایاب ہو جاتی ہیں، اور وہ شکار کے رویے میں انفرادی تغیرات ظاہر کر سکتی ہیں ۔ مزید برآں، بلیاں جانوروں کی لاشوں کو کھاتی ہیں، بشمول ان جانوروں کے جو وہ مار نہیں سکتیں، اور لوگوں کے چھوڑے ہوئے خراب اور ضائع شدہ کھانے کو کھاتی ہیں، جو انہیں ان وسائل کا استحصال کرنے اور مختلف ماحولیاتی نظاموں میں موجود رہنے کی اجازت دیتا ہے اور ممکنہ طور پر زیادہ کثافتوں پر۔ تاہم، بلیوں کے لیے کھانے کا غالب ذریعہ ان کی اعلی توانائی کی ضروریات کی وجہ سے لاشوں کو کھانا نہیں ہے, , ,۔

اگرچہ پچھلا کام کسی مخصوص براعظم (مثلاً، آسٹریلیا)، ماحولیاتی نظام کی قسم (مثلاً، جزائر)، یا ٹیکسونومک گروپ (مثلاً، چمگادڑ) کے اندر بلیوں کی خوراک کا جائزہ لینے پر مرکوز رہا ہے، ہماری معلومات کے مطابق بلیوں کے ذریعہ کھائی جانے والی انواع کی جامع دستاویز کرنے کی کوئی عالمی کوشش نہیں کی گئی ہے۔ نتیجتاً، ہمیں عالمی پیمانے پر یہ علم نہیں ہے کہ بلیاں مختلف جانوروں کے ٹیکسا کو کس حد تک کھاتی ہیں، بشمول تحفظ کے خدشات کی انواع، چاہے کوئی ٹیکسا ہو جسے بلیاں کھانے سے گریز کرتی ہیں، اور اگر بلیوں کے شکار یا لاشوں کے سائز کی بنیاد پر خوراک کی انتخابیت ہے۔ یہاں ہم ایک عالمی سطح پر تقسیم شدہ حملہ آور انواع کے بارے میں ان علمی خلا کو پُر کرتے ہیں جو بلیوں کی خوراک کے سب سے بڑے ڈیٹا بیس کی تعمیر اور تشخیص کرکے آج تک کی گئی ہے۔ خاص طور پر، ہم نے دنیا بھر میں آزاد گھومنے والی گھریلو بلیوں کی خوراک کو ٹیکسونومک گروپ، جزیرہ بمقابلہ براعظمی مقام، تحفظ کی حیثیت، خوراک کی انواع کے جسمانی وزن، اور غذائی اشیاء کو دستاویز کرنے کے لیے استعمال کیے گئے طریقہ کار کے لحاظ سے مقداری بنایا ۔

544 مطالعات میں سے، 533 نے ہماری انواع کی سطح کے ڈیٹا کے معیار کو پورا کیا اور ان میں 2084 انواع شامل تھیں جو بلیوں نے کھائی تھیں۔ خاص طور پر، یہ 2084 انواع بلیوں کی خوراک کا ایک محتاط تخمینہ فراہم کرتی ہیں جو انواع کے جمع ہونے کے منحنی خطوط پر مبنی ہیں، اس بات کی نشاندہی کرتے ہیں کہ جیسے جیسے مزید مطالعات کی جائیں گی، ہم بہت سی مزید شکار کی انواع کو دریافت کریں گے (Fig. [1] )۔ ہمیں معلوم ہوا کہ بلیوں کی خوراک پر شائع ہونے والے مطالعات کی تعداد پچھلی صدی کے دوران ڈرامائی طور پر بڑھ گئی ہے (Fig. [1] ), جن میں سے زیادہ تر مطالعات آسٹریلیا (قریبی جزائر سمیت؛n = 215) یا شمالی امریکہ (n = 144; Table [1] ) میں کی گئیں۔ نتیجتاً، ان دو علاقوں میں کی جانے والی مطالعات نے بھی بلیوں کے ذریعہ کھائی جانے والی انواع کی سب سے زیادہ تعداد میں حصہ ڈالا (Table [1] )۔ عالمی سطح پر، 288 مطالعات جزائر پر اور 237 براعظموں پر کی گئیں اور ان میں بالترتیب 797 اور 1,253 انواع کے ریکارڈ شامل تھے (Figure [SI1] )۔

بلیوں کے ذریعہ شکار یا لاشوں کو کھانے والی انفرادی انواع میں، پرندے 47.07% (981 انواع) پر مشتمل تھے، اس کے بعد رینگنے والے جانور (463 انواع، 22.22%)، ممالیہ (431 انواع، 20.68%)، کیڑے (119 انواع، 5.71%)، اور ایمفیبینز (57 انواع، 2.74%; Fig. [2] )۔ ہم نے باقی 6 طبقات (Actinopterygii, Arachnida, Chilopoda, Diplopoda, Gastropoda, Malacostraca;n = 33 انواع) کو "دیگر" میں ضم کر دیا جو مجموعی انواع کی تعداد کا 1.58% نمائندگی کرتا ہے (Fig. [2] , Table SI [1] )۔

براعظم کے لحاظ سے، پرندوں کی انواع کے ٹیکسونومک نمونوں میں عمومی رجحان جو سب سے زیادہ عام شکار ہیں، اس کے بعد رینگنے والے جانور، افریقہ، ایشیا، اور آسٹریلیا میں مستقل تھے (Fig. [2B] )۔ انٹارکٹیکا میں (دیکھیںMethods ), یورپ، شمالی امریکہ، اور جنوبی امریکہ میں، دوسرا سب سے عام شکار ممالیہ تھا (Fig. [2B] ), حالانکہ انٹارکٹیکا میں صرف پرندے اور ممالیہ شامل تھے۔ خاص طور پر، افریقہ میں کیڑے تیسرا سب سے عام شکار تھے اور جنوبی امریکہ میں "دیگر" زمرہ رینگنے والے جانوروں کے ساتھ تیسرے سب سے عام شکار کے لیے بندھا ہوا تھا۔

سب سے زیادہ عام طور پر دیکھے جانے والے شکار میں شامل ہیں گھریلو چوہا (Mus musculus ,n = 158 مطالعات, 29.64%), یورپی خرگوش (Oryctolagus cuniculus ,n = 114 مطالعات, 21.39%), کالا چوہا (Rattus rattus ,n = 74 مطالعات, 13.88%), گھریلو چڑیا (Passer domesticus ,n = 58 مطالعات, 10.88%), اور بھورا چوہا (Rattus norvegicus ,n = 54 مطالعات, 10.13%) (Table SI [2] ).

کل 347 (16.65%) بلیوں کے کھائے گئے انواع تحفظ کے لحاظ سے تشویش کا باعث ہیں، جو IUCN ریڈ لسٹ پر قریب خطرے میں، خطرے میں (یعنی، کمزور، خطرے میں یا شدید خطرے میں) یا معدوم کے طور پر درج ہیں (Fig. [3] ), ان میں دنیا کے 7.1% پرندے، 4.9% ممالیہ، اور 2.7% رینگنے والے جانور شامل ہیں (Table [2] )۔ عالمی سطح پر، جزیروں پر شناخت شدہ انواع میں سے 25.22% تحفظ کے لحاظ سے تشویش کا باعث ہیں، جبکہ براعظموں پر شناخت شدہ انواع میں سے صرف 8.62% تحفظ کے لحاظ سے تشویش کا باعث ہیں (Figure SI [1] )۔ براعظموں/قریب ترین براعظموں میں، زیادہ تر کھائی جانے والی انواع کو کم تشویش کے طور پر درجہ بندی کیا گیا ہے (افریقہ میں 61.99% سے یورپ میں 86.30%; Fig. [3] )۔ 347 تحفظ کے لحاظ سے تشویش کا باعث انواع میں، پرندے تمام زمروں میں سب سے زیادہ تعداد میں تھے، جن کے بعد ممالیہ اور رینگنے والے جانور تھے (Table [2] )۔ ہمیں آسٹریلیا، میکسیکو، ریاستہائے متحدہ امریکہ، اور نیوزی لینڈ سے بلیوں کے 11 انواع کے کھانے کے ریکارڈ ملے جو بعد میں جنگل میں معدوم (EW) یا معدوم (EX) کے طور پر درج کیے گئے، جن میں سے بہت سے جزیرے کے مقامی تھے (مثلاً، ہوائی کوا [Corvus hawaiiensis ]; Table [3] )۔ ہم نوٹ کرتے ہیں کہ 155 انواع، یا ہماری بلی کی غذائی تالیف میں کل تعداد کا 7.44% ابھی تک IUCN کے تحت تشخیص نہیں کی گئی ہیں اور ان میں سے زیادہ تر انواع (n = 108) کیڑے تھے۔ مزید برآں، ہم نوٹ کرتے ہیں کہ "تشخیص نہیں کی گئی" شکار کی انواع کا تناسب افریقہ (17.54%) اور ایشیا (10.64%) کے لیے نسبتاً زیادہ ہے۔

بلی کی غذا میں رپورٹ کی گئی فقاری انواع کا درمیانی بالغ جسمانی وزن تمام انواع کے لیے 45.45 گرام تھا، 13.67 گرام ایمفیبینز کے لیے، 62.42 گرام پرندوں کے لیے، 53.22 گرام ممالیہ کے لیے، اور 21.35 گرام رینگنے والے جانوروں کے لیے (Fig. [4] )۔ خاص طور پر، بلیاں تقریباً تمام سائز کی حدود میں شکار کھاتی ہیں (Fig. [4] )۔ ہر ٹیکسونومک کلاس میں کھائے جانے والے سب سے چھوٹے فقاری جانور عام بونے سکنک (Menetia greyii ; 0.47 گرام), جنوبی بھورا درخت مینڈک (Litoria ewingii ; 1.69 گرام), ایٹرسکن شریو (Suncus etruscus ; 2.37 گرام), اور روبی تھروٹڈ ہمینگ برڈ (Archilochus colubris ; 3.09) تھے۔ ہر کلاس کے لیے رپورٹ کی گئی سب سے بڑی فقاری انواع امریکی بیل مینڈک (Lithobates catesbeianus ; 467.80 گرام), ایمو (Dromaius novaehollandiae ;≈  34 کلوگرام), سبز سمندری کچھوا (Chelonia mydas ;≈  133 کلوگرام), اور گھریلو گائے (Bos taurus ;≈  760 کلوگرام) تھے۔ ہم نے غذا کے جسمانی وزن میں کوئی واضح حد نہیں دیکھی، حالانکہ ریکارڈ کی گئی بہت سی بڑی انواع ممکنہ طور پر نابالغوں کے طور پر شکار کی گئی تھیں یا بالغوں کے طور پر کھائی گئی تھیں۔

غذا کا تعین کرنے کے لیے استعمال کیے گئے طریقوں میں سب سے زیادہ تعداد میں انواع کی شناخت مشاہدہ شدہ شکار کے ذریعے کی گئی، جبکہ مفروضہ شکار کے نتیجے میں سب سے کم تعداد میں انواع کی شناخت ہوئی (Fig. SI [2] )۔ خاص طور پر، شکار کا براہ راست مشاہدہ بہت سے مطالعات میں بڑھ رہا ہے، ممکنہ طور پر کیمرہ ٹریپس اور جانوروں پر مبنی ویڈیو کے اطلاق کے ساتھ۔ طریقہ کار کا نقطہ نظر دلچسپی کے ٹیکسونومک گروپ کے لحاظ سے مختلف ہوتا ہے، اس طرح کوئی واحد طریقہ نہیں تھا جو مستقل طور پر سب سے عام طریقہ تھا جس کے ذریعے غذا کا تعین کیا گیا تھا (Figure [SI2] )۔

ہمارا مطالعہ دنیا کے سب سے کامیاب اور وسیع پیمانے پر پھیلے ہوئے حملہ آور شکاریوں کی شکاری عادات پر روشنی ڈالتا ہے۔ ہماری معلومات کے مطابق یہ مطالعہ بلی کی غذا کی اب تک کی سب سے جامع عالمی ترکیب ہے، اور ممکنہ طور پر دنیا بھر میں کسی بھی نوع کے لیے سب سے بڑا ہے۔ مثال کے طور پر، کارنی ڈائیٹ ڈیٹا بیس میں صرف 823 شکار/غذا کی انواع شامل ہیں جو 103 ممالیہ گوشت خوروں کے درمیان تشخیص کی گئی ہیں، جن میں چیتے (Panthera pardus ) سب سے زیادہ انواع کی تنوع کھاتے ہیں (n = 214)۔ اس طرح، ہمارا تجزیہ بلیوں کے ساتھ ساتھ گوشت خوروں کے شکار اور کھائی جانے والی انواع کی تعداد کے پچھلے تخمینوں میں نمایاں اضافہ کرتا ہے اور اس طرح ایک عالمی بنیاد قائم کرتا ہے۔ بلی کی غذا کے پچھلے تخمینوں میں دنیا بھر کے جزیروں پر 248 فقاری اور غیر فقاری انواع اور آسٹریلیا میں 833 فقاری انواع شامل تھیں, , , ,۔ کل ملا کر، ہم نے 2084 انواع کی شناخت کی جو بلیوں کے ذریعے شکار کی گئیں، کھائی گئیں، یا کسی اور طرح سے کھائی گئیں، جن میں تقریباً 9% معروف پرندے (11,162 بیان کردہ پرندوں کی بنیاد پر)، 6% سے زیادہ معروف ممالیہ (6596 بیان کردہ ممالیہ کی بنیاد پر)، اور تقریباً 4% معروف رینگنے والے جانور (11,733 بیان کردہ رینگنے والے جانوروں کی بنیاد پر) شامل ہیں۔ کھائی جانے والی انواع میں سے، 347 (16.65%) کو IUCN ریڈ لسٹ پر قریب خطرے میں یا اس سے زیادہ تشویش کے طور پر درج کیا گیا تھا۔ جبکہ بلی کی غذا کے مطالعات کو عالمی سطح پر تقسیم کیا گیا پایا گیا، زیادہ تر آسٹریلیا اور شمالی امریکہ سے ہیں، جبکہ افریقہ، یوریشیا کے کچھ حصے، اور جنوبی امریکہ سبھی کم نمائندگی والے ہیں۔ اس طرح، اس بات کا قوی امکان ہے کہ جیسے جیسے کم نمائندگی والے مقامات اور غیر تشخیص شدہ جزیروں میں مطالعات کیے جاتے ہیں، تحفظ کے لحاظ سے تشویش کا باعث مزید انواع ہماری فہرست میں شامل کی جائیں گی ۔

مجموعی طور پر، یہ دریافت کہ بلیاں بڑی تعداد میں پرندوں کی انواع کھاتی ہیں (981 انواع) حیران کن نہیں ہے کیونکہ عالمی سطح پر پرندوں کی انواع ممالیہ سے زیادہ ہیں (~11,000 بمقابلہ ~6500 ممالیہ)، اور یہ کہ بلیوں کے زیر قبضہ جزیرے کے ماحولیاتی نظاموں میں پرندے موجود ہیں جو مقامی ممالیہ سے خالی ہیں۔ خاص طور پر، عالمی سطح پر 11,000 سے زیادہ رینگنے والے جانوروں کی انواع ہیں، لیکن بلیوں کے ذریعہ کھائی جانے والی رینگنے والے جانوروں کی تعداد ممالیہ کی طرح تھی (بالترتیب 463 اور 431)۔ یہ فرق اس صورت میں ہو سکتا ہے اگر بلیاں رینگنے والے جانوروں کے مقابلے میں ممالیہ اور پرندوں کو ترجیح دیتی ہیں، لیکن شاید زیادہ امکان ہے کہ نمونے لینے کے تعصبات کی وجہ سے، بشمول غذا کے نمونوں میں رینگنے والے جانوروں کی کم شناخت، پرندوں اور ممالیہ کے مقابلے میں بہت سے رینگنے والے جانوروں کے چھوٹے رینج سائز، اور یہ کہ رینگنے والے شکار کے مطالعات نے دوسرے رینگنے والے جانوروں کے ذریعہ اینڈو تھرمز کے مقابلے میں زیادہ سطحوں پر شکار کی اطلاع دی ہے ۔ جو حیران کن ہے وہ غذا میں شناخت شدہ ایمفیبینز اور کیڑوں کی نسبتاً زیادہ تعداد ہے، خاص طور پر اس بات کو مدنظر رکھتے ہوئے کہ بہت سے مطالعات میں کیڑوں کی مکمل ٹیکسونومک ریزولوشن نہیں ہے۔ خاص طور پر، ایمفیبین اور غیر فقاری جلد اور ایکسو اسکیلٹن کی نوعیت کی وجہ سے، بالترتیب، بکھرے ہوئے یا ہضم شدہ باقیات کا جائزہ لینے کے لیے اکثر جسمانی مواد بہت کم ہوتا ہے,۔ مزید برآں، بلی کی غذا کے مطالعات میں غیر فقاری ٹیکسونومسٹ کی شمولیت بہت کم ہے۔ بلی کی غذا میں پائی جانے والی انواع کی تنوع سے جو بات واضح ہے وہ یہ ہے کہ بلیاں ان انواع کے بڑے حصے کو شکار کرتی ہیں اور کھاتی ہیں جو ان مناظر میں دستیاب ہیں جن میں وہ چارہ لیتی ہیں اور یہ کہ وہ جسمانی وزن کے اشارے کے طور پر تمام انواع کی تقسیم کی نمائندگی کرتی ہیں (Fig.[4])۔ بلیاں بنیادی طور پر وہی کھا رہی ہیں جو موجود ہے اور اگر کوئی نوع غذا کے تجزیے میں غائب ہے تو یہ ممکنہ طور پر شکار یا تو غیر موجود ہے یا آس پاس کے ماحول میں نایاب ہے، بلیوں کے لیے پکڑنا مشکل ہے اور اس لیے کم منافع بخش ہے، یا شکار کو بکھرے ہوئے یا ہضم شدہ مطالعات میں شناخت کرنا مشکل ہے (مثلاً، غیر فقاری)۔ خاص طور پر، شناخت کے مسئلے میں مستقبل میں کمی آ سکتی ہے کیونکہ بکھرے ہوئے تجزیے میں مالیکیولر طریقے استعمال کیے جاتے ہیں ۔

جیسا کہ بہت سے دوسرے منظم ماحولیاتی مطالعات میں پایا گیا ہے، ہمارے نتائج میں جغرافیائی تعصبات ہیں۔ جزیرے کے نظاموں اور آسٹریلیا کو چھوڑ کر، بلی کی غذا کے مطالعات جہاں کیے گئے ہیں اور دنیا کے حیاتیاتی تنوع کے ہاٹ اسپاٹس، خاص طور پر افریقہ، ایشیا، اور جنوبی امریکہ میں، کے درمیان نسبتاً کم اوورلیپ ہے۔ ان مقامات سے مطالعات کی کمی تحقیق کی کوششوں پر معاشی پابندیوں کی وجہ سے ہو سکتی ہے، لیکن ممکنہ طور پر بلیوں کے مقامی جنگلی حیات پر اثرات کے بارے میں کم تشویش بھی ہو سکتی ہے۔ جزیرے اور آسٹریلیا اچھی طرح سے مطالعہ کیے گئے ہیں جزوی طور پر ان مقامی انواع پر بلیوں کے اثرات کے بارے میں تشویش کی وجہ سے جو بلی کے شکاری کے خطرے سے ناواقف ہیں, , ,۔ اس کے برعکس، افریقہ، ایشیا، اور جنوبی امریکہ میں بہت سی مقامی بلی کی انواع ہیں، جس کا مطلب یہ ہو سکتا ہے کہ i) مقامی شکار بلیوں کے شکار کے لیے کم حساس ہیں کیونکہ ان کی بلیوں کے ساتھ ہم ارتقاء ہے، اور ii) بلیوں کے لیے مقامی گوشت خوروں کے ساتھ مقابلہ کی وجہ سے کم کامیاب ہونا ممکن ہے ۔ درحقیقت، جزائر پر خطرے سے دوچار شکار کی انواع کا تناسب (25.22%) براعظموں (8.62%) کے مقابلے میں بہت زیادہ ہے، شاید شکار اور شکاریوں کے درمیان ہم ارتقاء کی تاریخ کی کمی، جزائر پر عالمی سطح پر پائے جانے والے مقامی انواع کی اعلی شرح، اور جزائر تک محدود خطرے سے دوچار انواع کی غیر متناسب زیادہ تعداد کی وجہ سے ۔ اس کے باوجود، سیارے کے بہت سے سب سے زیادہ حیاتیاتی مقامات میں نمونے لینے کی کمی سے پتہ چلتا ہے کہ ہم ممکنہ طور پر بلیوں کے شکار اور کھانے کے جائزے کے لیے اہم مقامات کو کھو رہے ہیں۔

جبکہ بلی کی غذا میں پائی جانے والی سب سے زیادہ عام انواع کاسموپولیٹن غیر مقامی انواع ہیں (مثلاً،Mus musculus ,Rattus rattus ), زیادہ تر مطالعات میں رپورٹ کی گئی انواع کا کوئی بڑا مجموعہ نہیں تھا۔ درحقیقت، بلی کی غذا پر زیادہ تر نئے مطالعات نے اب تک غیر ریکارڈ شدہ انواع کو جنم دیا۔ نتیجتاً، جبکہ بہت سے مطالعات میں چھوٹے حملہ آور چوہے اور پرندے پائے جاتے ہیں، ہمیں بلیوں کے صرف ایک چھوٹے سے عام پرندوں اور ممالیہ کے مجموعے کو کھانے کا کوئی اشارہ نہیں ملا، اور بہت سے پچھلے مطالعات نے اطلاع دی ہے کہ یہاں تک کہ ان مقامات پر بھی جہاں بلیاں حملہ آور کیڑوں کی انواع کا زیادہ تناسب کھاتی ہیں، بہت سی مقامی انواع بھی کھائی جاتی ہیں ۔ بلکہ، اور خاص طور پر جزائر، حیاتیاتی تنوع کے ہاٹ اسپاٹس، اور دیگر علاقوں میں جہاں اعلی مقامی انواع ہیں، ہم نے پایا کہ بلیوں نے زیادہ عام طور پر ان انواع کا شکار کیا جو دستیاب تھیں۔

پچھلی تحقیق کا مرکز ان انواع پر بلیوں کے شکار پر رہا ہے جو تحفظ کے لحاظ سے اہم ہیں, ,۔ بلیوں کے شکار اور مردار خوری کی عالمی رپورٹس کو جمع کرکے، ہمارے نتائج IUCN ریڈ لسٹ کی انواع کی بڑی تعداد، خاص طور پر پرندے اور ممالیہ، جو بلیوں کے ذریعے کھائے جاتے ہیں اور ممکنہ طور پر متاثر ہوتے ہیں، کی ایک مکمل تصویر پیش کرتے ہیں۔ بلیوں کی خوراک میں تحفظ کے لحاظ سے اہم انواع کی زیادہ نمائندگی (16.65%) تشویشناک ہے کیونکہ بلیوں کو پہلے ہی عالمی سطح پر 26% پرندوں، ممالیہ، اور رینگنے والے جانوروں کی معدومیت سے جوڑا گیا ہے، اور انہیں بہت سی موجودہ خطرے میں پڑی انواع کے لیے بڑے خطرے کے طور پر تسلیم کیا گیا ہے ۔ چونکہ ہم نے تحفظ کے لحاظ سے اہم انواع کی قومی یا علاقائی درجہ بندیوں کو نہیں دیکھا، نہ ہی ذیلی انواع یا مخصوص آبادی کے حصوں کو، یہ ممکن ہے کہ ہماری فہرست میں مقامی سطح پر تشویش کی بہت سی اضافی اقسام شامل ہوں۔ مزید برآں، یہ دیکھتے ہوئے کہ بہت سی انواع، خاص طور پر غیر فقاری، IUCN کے ذریعہ نہیں جانچی گئیں اور/یا انہیں خوراکی مطالعات میں اس سطح تک نہیں پہچانا گیا کہ ان کی جانچ کی جا سکے، اضافی خطرے میں پڑی انواع ممکنہ طور پر ہماری مطالعات میں بلیوں کی خوراک میں موجود ہیں۔

جبکہ ہماری تجزیہ میں بلیوں کے ذریعہ کھائی جانے والی انواع کی تعداد پہلے سے بیان کردہ سے زیادہ ہے، ہماری انواع کے جمع ہونے کے منحنی خطوط کی تمام تبدیلیاں کوئی اسیمپٹوٹی نہیں دکھاتی ہیں، جو اس بات کی نشاندہی کرتی ہیں کہ اضافی مطالعات ممکنہ طور پر ان گنتیوں کو بڑھانے کے لیے جاری رہیں گی جو ہم رپورٹ کرتے ہیں۔ یہ محتاط تخمینہ چند وجوہات کی بنا پر ممکنہ طور پر درست ہے۔ سب سے پہلے، اس مطالعے کے مقاصد کے لیے، ہم نے صرف ان مطالعات پر غور کیا جن میں شکار یا مردار خوری کو دستاویزی شکل دی گئی تھی اور کھپت کی کوئی بھی کہانی یا دوسرے ہاتھ کی رپورٹس ریکارڈ نہیں کی گئیں۔ دوسرا، یہاں تک کہ ان مطالعات میں جو ہمارے ثبوت کے معیار پر پورا اترتے ہیں، بہت سے چھوٹے ممالیہ، کیڑے، اور رینگنے والے جانوروں کے لیے حیاتیاتی درجہ بندی کی وضاحت مستقل طور پر جنس یا ترتیب سے آگے نہیں کی گئی تھی کیونکہ ایسی انواع کی شناخت میں دشواری ہوتی ہے۔ تیسرا، تقریباً تمام مطالعات میں بہت سی انواع شامل تھیں جنہیں محض 'دیگر' یا 'نامعلوم' کے طور پر جمع کیا گیا تھا۔ چوتھا، دنیا کے بڑے علاقوں کو بلیوں کے شکار کے لیے نمونہ نہیں بنایا گیا ہے، بشمول بہت سے استوائی علاقے، حیاتیاتی تنوع کے گرم مقامات، اور دیگر حیاتیاتی مقامات۔ پانچواں، خوراک کے ڈیٹا کو جمع کرنے کے مختلف طریقے بلیوں کے کھانے کی مختلف نمائندگی اور تعصبات فراہم کرتے ہیں، شکار-گھر-لائے جانے والے طریقے چھوٹے شکار کی کم نمائندگی کرتے ہیں اور شکار-کھائے جانے والے طریقے غیر خوشگوار شکار کو ریکارڈ کرنے کا امکان کم ہوتا ہے۔ چھٹا، موقع پرست شکاریوں کے طور پر، بلیاں جانوروں کو مار دیتی ہیں جنہیں وہ پھر نہیں کھاتیں اور ایسی اموات کو بلیوں کے شکار کی تقریباً تمام رپورٹنگ میں نظر انداز کر دیا جاتا ہے۔ اس طرح، ہم توقع کرتے ہیں کہ ہمارا ڈیٹا بیس مستقبل میں نمایاں طور پر بڑھے گا اور عالمی سطح پر بلیوں کے ذریعہ کھائی جانے والی انواع کی حقیقی مقدار کا صرف ایک حصہ پیش کرتا ہے ۔

اگرچہ ہماری تحقیق کا مرکز بلیوں کی خوراک پر تھا، یہ 150 سال سے زیادہ کے ادب پر مبنی ہے جو ماحول پر آزادانہ گھومنے والی بلیوں کے منفی اثرات کو دستاویزی شکل دیتا ہے۔ شکار کے علاوہ، ان اثرات میں متعدد بلیوں سے پیدا ہونے والی بیماریاں شامل ہیں جو جنگلی حیات اور انسانی صحت اور بہبود کو متاثر کرتی ہیں، بشمول ٹاکسوپلاسموسس، طاعون، اور ریبیز، اور کچھ علاقوں میں (جیسے آسٹریلیا)، ان میں سے کچھ بیماریاں بلیوں کے بغیر نہیں ہوتیں۔ مزید برآں، آزادانہ گھومنے والی بلیاں جو گروہوں (عرف کالونیاں) میں رہتی ہیں ان مسائل کو بڑھا سکتی ہیں اور اضافی مسائل پیش کر سکتی ہیں جن میں اضافی غذائی اجزاء کی لوڈنگ، صفائی، اور جنگلی حیات کے تنازعات شامل ہیں۔ آخر میں، بلیوں کی صرف موجودگی باہر خوف کے مناظر پیدا کر سکتی ہے جو جنگلی حیات کے رویے میں تبدیلیوں کا نتیجہ بنتی ہے کہ ایک نوع زمین کی تزئین پر کہاں ہوتی ہے ان کے چارہ تلاش کرنے کے فیصلوں اور افزائش کی کامیابی کے لیے، جو خطرے میں پڑی انواع کے لیے خاص تشویش کا باعث ہے ۔ مجموعی طور پر، یہ اثرات آزادانہ گھومنے والی بلیوں کے اثرات کو کم کرنے کی کوشش کرنے والے پالیسی اور انتظامی اقدامات کو آگے بڑھانے کے لیے مضبوط محرک فراہم کرتے ہیں۔

مجموعی طور پر، ہمارے نتائج ظاہر کرتے ہیں کہ بلیاں بلا امتیاز شکاری ہیں اور بنیادی طور پر کسی بھی قسم کے جانور کو کھا لیتی ہیں جسے وہ زندگی کے کسی مرحلے پر پکڑ سکتی ہیں یا مردار خوری کر سکتی ہیں۔ یہ خوراک کی وسعت مزید ثبوت فراہم کرتی ہے کہ بلیاں کس طرح (یا کر سکتی ہیں) مقامی انواع کے ساتھ تعامل کر سکتی ہیں اور ماحولیاتی نظام کو درہم برہم کر سکتی ہیں کیونکہ وہ کسی ایک غذائی سطح یا حیاتیاتی گروپ پر انحصار نہیں کرتی ہیں۔ نتیجتاً، بلیاں انواع کے تعاملات کے ایک وسیع تر سیٹ کو متاثر کر رہی ہیں جتنا پہلے سمجھا گیا تھا۔ آخر کار، اگرچہ ہمارے نتائج محتاط ہیں، وہ اس حد کو اجاگر کرتے ہیں جس میں ایک وسیع پیمانے پر تقسیم شدہ حملہ آور نوع دنیا بھر میں انواع کے ساتھ تعامل کر رہی ہے، جو تحفظ، انتظام، اور پالیسی کے کام کو آگے بڑھانے کے لیے اہم معلومات ہے۔

ہم نے گوگل اسکالر اور ویب آف سائنس میں 'بلیوں کا شکار'، 'جنگلی بلی'، 'بلیوں کی خوراک'، اور 'Felis catus' کے کلیدی الفاظ کا استعمال کرتے ہوئے دونوں ہم مرتبہ جائزہ شدہ اور سرمئی ادب کی تلاش کے ذریعے بلیوں کی خوراک پر معلومات مرتب کیں۔ ہر ذریعہ سے ہم نے اس کا جائزہ لیا کہ آیا اس میں بلیوں کی خوراک یا شکار کے بارے میں ڈیٹا موجود ہے، نیز بلیوں کی خوراک اور شکار سے متعلق اضافی منفرد مضامین یا ڈیٹا بیس کے لیے اس کے حوالہ جات کے سیکشن کا جائزہ لیا۔ ہم نے حوالہ جات میں شناخت کیے گئے کسی بھی نئے ذرائع کا بار بار جائزہ لیا تاکہ دونوں ڈیٹا اور اضافی مضامین یا ڈیٹا بیس کو مکمل طور پر جانچا جا سکے (یعنی، ہر مخطوطہ کے حوالہ جات کو پایا گیا اور بلیوں سے متعلق ہونے کی صورت میں خوراک کے لیے جانچا گیا)۔ اس مکمل تلاش کے بعد ہم نے 544 منفرد اشاعتوں (کتابیں، جرنل مضامین، تھیسس، اور ایجنسی رپورٹس) کی شناخت کی جن میں بلیوں کی خوراک پر ڈیٹا موجود تھا۔ وہ ریکارڈز جو انواع کی سطح پر شناخت نہیں کیے گئے تھے (یعنی، صرف جنس تک) کو تجزیہ سے ہٹا دیا گیا، جس سے ہمارے پاس بلیوں کے ذریعہ کھائی جانے والی انواع کے ریکارڈز کے ساتھ 533 منفرد اشاعتیں رہ گئیں۔ اس تحقیق کے مقاصد کے لیے، جب ریکارڈز کو ذیلی انواع کی سطح پر شناخت کیا گیا، تو ہم نے انہیں انواع کی سطح پر دوبارہ درجہ بندی کیا۔ یہ ڈیٹا بیس مئی 2021 تک دستیاب تمام ادب پر مشتمل ہے۔

کسی اشاعت میں درج ہر نوع کے لیے ہم نے نوع کا عام اور/یا سائنسی نام، مطالعہ کا مقام، خوراک کے ڈیٹا کی شناخت/جمع کرنے کے لیے استعمال ہونے والا طریقہ (مثلاً، فضلہ تجزیہ، آنتوں کے مواد کا تجزیہ، مشاہدہ شدہ شکار)، خوراک/شکار کے نمونے جمع کیے جانے کے سال، اور اشاعت کا حوالہ ریکارڈ کیا۔ ہم نے کسی بھی پرانے نوع کے نام کو اپ ڈیٹ کیا اور اس کی مکمل حیاتیاتی درجہ بندی شامل کی (یعنی، شعبہ، طبقہ)، اس بات کی نشاندہی کی کہ آیا مطالعہ کا مقام جزیرے پر تھا یا براعظم پر، قریب ترین براعظم، اور اسے IUCN ریڈ لسٹ میں تحفظ کی حیثیت کے مطابق درجہ بندی کیا ۔ نوٹ کریں کہ اگرچہ انٹارکٹیکا کے براعظم پر کوئی نوع ریکارڈ نہیں کی گئی تھی، لیکن ایسے جزیرے ہیں جو جغرافیائی طور پر انٹارکٹیکا کے قریب ہیں دوسرے براعظموں کے مقابلے میں اور اس طرح یہاں انٹارکٹک کے طور پر درج ہیں۔ اس کے علاوہ، ہم نے پرندوں، ممالیہ، رینگنے والے جانوروں، اور ایمفیبینز کے لیے بالغ جسمانی وزن کے اوسط ڈیٹا کو شامل کیا ۔

شکار کی انواع کی دولت میں ہر مطالعہ کے تعاون کا جائزہ لینے کے لیے ہم نے انواع کے جمع ہونے کے منحنی خطوط کا استعمال کیا۔ خاص طور پر، ہم نے ہر اشاعت (یعنی، مطالعہ) کو ایک منفرد مطالعہ 'سائٹ' کے طور پر سمجھا اور پروگرام R (ورژن 4.1.0) میں ویگن پیکیج (ورژن 2.5-7) کا استعمال کرتے ہوئے فی مطالعہ شامل کی گئی منفرد انواع کی کل تعداد کو شمار کیا۔ تاہم، ان ریکارڈز کو تجزیہ کے مقاصد کے لیے منفرد مشاہدات کے طور پر صرف اس صورت میں سمجھا گیا جب اس نوع کی پہلے ڈیٹا بیس میں شناخت نہیں کی گئی تھی۔ ہم نے ویگن پیکیج میں specaccum کمانڈ (طریقہ = 'عین مطابق') کا استعمال کرتے ہوئے انواع کے جمع ہونے کے منحنی خطوط کو (1) تمام مطالعات، (2) براعظموں اور جزیروں پر کیے گئے مطالعات (ان میں شامل نہیں جو 'براعظم اور جزیرہ' کے طور پر درجہ بندی کیے گئے ہیں)، اور (3) آسٹریلیا اور شمالی امریکہ میں کیے گئے مطالعات کے لیے پیدا کیا۔ ہم نے آسٹریلیا اور شمالی امریکہ کے لیے انواع کے جمع ہونے کے منحنی خطوط شامل کیے کیونکہ وہ ہمارے ڈیٹا بیس میں زیادہ تر مطالعات پر مشتمل تھے۔ مزید برآں، ہم توقع کرتے ہیں کہ اگر ان علاقوں میں اسیمپٹوٹی پوری نہیں ہوئی تھی جہاں دیگر علاقوں کے مقابلے میں نمایاں طور پر زیادہ مطالعات کیے گئے تھے، تو اسیمپٹوٹی کسی دوسرے براعظم پر پوری نہیں ہوئی ہوگی ۔
