ویرایش ژن برای عفونت ویروس هرپس سیمپلکس نهفته کاهش بار ویروسی و ریزش در محیط زنده

چکیده: درمان‌های ضد ویروس هرپس سیمپلکس فقط سرکوبگر هستند زیرا ویروس نهفته هرپس سیمپلکس را در نورون‌های گانگلیونی، منبع بیماری tái‌발، از بین نمی‌برند. ما یک روش بالقوه درمانی علیه عفونت ویروس هرپس سیمپلکس بر اساس ویرایش ژن با استفاده از مگا نوکلئازهای خاص ویروس هرپس سیمپلکس تحویل داده شده توسط وکتورهای ویروس مرتبط با آدنو (AAV) توسعه داده‌ایم. ویرایش ژن انجام شده با دو مگا نوکلئاز ضد ویروس هرپس سیمپلکس-1 تحویل داده شده توسط ترکیبی از AAV9، AAV-Dj/8 و AAV-Rh10 می‌تواند 90٪ یا بیشتر از DNA ویروس هرپس سیمپلکس نهفته را در مدل‌های موش از عفونت اوروفاسیال و تا 97٪ از DNA ویروس هرپس سیمپلکس نهفته در مدل‌های موش از عفونت ژنیتال از بین ببرد. با استفاده از یک روش دارویی برای فعال کردن مجدد ویروس هرپس سیمپلکس-1 نهفته، ما نشان می‌دهیم که کاهش بار ویروسی گانگلیونی منجر به کاهش قابل توجهی در ریزش ویروسی در موش‌های ماده درمان شده می‌شود. در حالی که این درمان به خوبی تحمل می‌شود، در برخی موارد، ما hepatotoxicity را در دوزهای بالا و شواهد هیستولوژیک ظریف از آسیب عصبی بدون علائم یا نقص‌های عصبی قابل مشاهده مشاهده می‌کنیم. ساده‌سازی رژیم از طریق استفاده از یک سرotype واحد (AAV9) که یک مگا نوکلئاز واحد را به یک منطقه تکراری ژنوم ویروس هرپس سیمپلکس هدف قرار می‌دهد، کاهش دوز و استفاده از یک پروموتور خاص نورون هر کدام منجر به تحمل پذیری بهبودیافته در حالی که کارایی را حفظ می‌کند. این نتایج پتانسیل درمانی ویرایش ژن برای بیماری ویروس هرپس سیمپلکس را تقویت می‌کند.

عفونت‌های HSV می‌توانند باعث ایجاد زخم‌های مکرر در ناحیه اوروفاسیال، کورنه، آنوژنیتال یا سایر نواحی شوند و عفونت نوزادان می‌تواند منجر به بیماری منتشر و عوارض عصبی شدید شود. عفونت ژنیتال با HSV-2 خطر ابتلا به HIV را افزایش می‌دهد و یکی از عوامل اصلی همه‌گیری جهانی HIV است. درمان‌های ضد ویروسی فعلی می‌توانند اپیزودهای حاد و سرکوب را درمان کنند، اما عفونت‌های تاسیس‌شده را درمان نمی‌کنند, , , ,. شیوع مجدد ناشی از توانایی HSV در ایجاد عفونت نهفته در نورون‌های گانگلیونی است که نواحی تحت تأثیر را عصب‌رسانی می‌کنند. عفونت نهفته HSV در گانگلیا تحت تأثیر داروهای ضد ویروسی سنتی قرار ندارد، که این عدم توانایی در درمان را توضیح می‌دهد و реакتیواسیون‌ها معمولاً پس از停止 درمان دوباره آغاز می‌شوند.

یک استراتژی بالقوه درمانی شامل ویرایش ژن در HSV نهفته است. در یک مطالعه اخیر، مگا نوکلئازهای تحویل‌شده توسط AAV بیش از 90٪ از ژنوم‌های HSV-1 را از گانگلیای سرویکال فوقانی موش‌های آلوده به HSV حذف کردند. با وجود کاهش قابل توجه بار ویروسی گانگلیونی پس از ویرایش ژن، ارتباطی که چنین کاهش می‌تواند برای عفونت HSV در انسان داشته باشد، نامعلوم است. افراد آلوده معمولاً با بار ویروسی گانگلیونی به خودی خود سروکار ندارند، بلکه در مورد بیماری‌های علامت‌دار و/یا ریزش ویروسی و خطر انتقال به دیگران هستند. مدل‌های موش موجود در توانایی خود برای حل این مسائل محدود هستند، زیرا موش‌های آلوده به HSV به ندرت HSV را به صورت خود به خود реакتیو می‌کنند یا ریزش ویروسی در بافت‌های محیطی نشان می‌دهند.

در اینجا، ما یک مدل از القای مولکول کوچک از HSV-1 را معرفی می‌کنیم و نشان می‌دهیم که ویرایش ژن یک کاهش دراماتیک نه تنها در بار ویروسی گانگلیونی، بلکه در ریزش ویروسی القا شده ایجاد می‌کند. بهینه‌سازی رویکرد درمانی از طریق ساده‌سازی رژیم، کاهش دوز و محدودیت بیان مگا نوکلئاز در بافت‌ها منجر به حذف تقریباً کامل اثرات ناخواسته بر کبد و گانگلیا می‌شود و از توسعه بالینی این استراتژی حمایت می‌کند.

ما قبلاً چندین سرotype AAV را برای تحویل مگا نوکلئاز به موش‌های آلوده به HSV ارزیابی کردیم و بهترین نتایج را با AAV-Rh10، به دنبال AAV8 و AAV1 به دست آوردیم. برای بهبود بیشتر کارایی، ما سرotype‌های AAV عصبی 추가‌ای را آزمایش کردیم، از جمله AAV7، AAV9، AAV-DJ و AAV-DJ/8، برای تحویل مگا نوکلئاز ضد HSV-1، m5، با دوز 10 ژنوم AAV (vg) در هر موش (Fig. [S1a] )، با استفاده از مدل بیماری HSV اوروفاسیال ما. هر دو AAV9 و AAV-Dj/8 نسبت به 10 vg AAV-Rh10، بهترین سرotype‌های قبلی ما، برتری داشتند و کاهش HSV را در گانگلیای سرویکال فوقانی (SCG) به ترتیب 95٪ (p = 10) و 90٪ (p = 0.018) نشان دادند، نسبت به کنترل‌های بدون درمان (Fig. [S1b] )، که با کاهش 65٪ قبلی که با m5 تنها با AAV-Rh10,به دست آمد، مقایسه می‌شود. به همین ترتیب، AAV9 و AAV-Dj/8 فعالیت بهتری نسبت به AAV-Rh10 در گانگلیای سه‌تایی (TG) نشان دادند، با کاهش بار ویروسی HSV به ترتیب 48٪ (p = 0.07) و 41٪ (p = 0.5) (Fig. [S1b] )، در مقایسه با مشاهده قبلی ما از کاهش قابل تشخیص با استفاده از AAV-Rh10 که m5 را تحویل می‌دهد. مسیر تحویل AAV (ورید خلفی چشم در مقابل داخل پوستی به پد عضله) هیچ影响 قابل تشخیصی بر کارایی انتقال یا ویرایش ژن AAV نداشت (Fig. [S1e–g] ).

ما قبلاً نشان دادیم که ویرایش ژن HSV را می‌توان با استفاده از ترکیباتی از سرotype‌های AAV برای تحویل مگا نوکلئاز، به جای یک سرotype AAV، افزایش داد، که این یافته را به ناهمگونی زیرمجموعه‌های عصبی در گانگلیای آلوده به HSV نسبت دادیم. بنابراین، ما ویرایش ژن را با مگا نوکلئاز ضد HSV-1، m5، که یک توالی در ژنUL19 را که برای پروتئین کپسید عمده VP5 کد می‌کند، برش می‌دهد،當 آن را با استفاده از سرotype‌های AAV تک یا ترکیبی از AAV9، AAV-Dj/8 و AAV-Rh10 (Fig. [S2a] ) به عنوان دوز کل 10 vg در هر موش تحویل دادیم. با توافق با نتایج قبلی ما، ترکیباتی از سرotype‌های AAV منجر به ویرایش ژن HSV قوی شد، با ترکیب سه‌گانه AAV9، AAV-Dj/8 و AAV-Rh10 که کاهش‌های especialmente قوی در بار ویروسی HSV و جهش‌زایی HSV باقی‌مانده در هر دو SCG و TG (Fig. [S2b–g] ) نشان داد.











همانطور که در بالا ذکر شد، عفونت HSV ژنیتال یک علت اصلی موربیدیتی در انسان است. بنابراین، ما درمان دوگانه مگا نوکلیاز با دوز کاهش‌یافته (دوز کل 1.8 × 10 vg/animal) را در موش‌های آلوده به HSV از طریق واژن ارزیابی کردیم (Fig. [6a] ). مطابق با نتایج قبلی ما، درمان با دوز کاهش‌یافته منجر به کاهش 78.8٪ (p = 0.02) تا 95.6٪ (p = 0.006) در ژنوم‌های ویروسی نهفته در DRGs شد (Fig. [6b] ).

سپس ما تلاش کردیم تا ارزیابی کنیم که آیا JQ1 می‌تواند باعثshedding HSV در مدل عفونت ژنیتال شود، همانطور که قبلا در مدل عفونت اکلر مشاهده کردیم. در طول 3 واکنش مجدد JQ1، فقط 2 از 8 حیوان کنترل (و 1 از 8 حیوان درمان شده با AAV/meganuclease) ویروس قابل تشخیصی را دفع کردند، نرخی پایین‌تر از 40-50٪ واکنش مجدد که معمولا بعد از عفونت اکلر مشاهده می‌شود (Fig. [6c, d] ). نرخ واکنش مجدد پایین‌تر مشاهده شده در مدل واژن در مقایسه با مدل اکلر ممکن است به دلیل سطح پایین‌تر بار ویروسی گانگلیونی در DRG باشد (10-10 vg/10 سلول در DRG، Fig. [6b] در مقابل 10-10 vg/10 سلول در SCG یا TG، Fig. [5b, c] ). در حالی که این نرخ واکنش مجدد پایین‌تر مانع از تجزیه و تحلیل آماری معنا دار شد، مشاهده اینکه 2 از 8 موش کنترل در طول 2 تا 3 روز متوالی ویروس دفع کردند، در حالی که فقط 1 از 8 موش درمان شده با AAV ویروس را در یک روز و در سطح پایین‌تر دفع کرد، از نظر کیفی با مشاهدات ما بعد از عفونت اکلر مطابقت دارد (Fig. [6c-g] ).

است دلیل کاهش کمتری بار ویروسی در موش های درمان شده با AAV9-E/CamKII-m4 در مقایسه با موش های درمان شده با AAV9-CBh-m4 باشد (Fig. [9f, g] ). ما نتیجه گرفتیم که رژیم AAV9-E/CamKII-m4 کارایی خود را حفظ می کند و تحمل پذیری بهتری در مقایسه با AAV9-CBh-m4 دارد.

آلودگی انسان به HSV طولانی است و در حالی که روش های ضد ویروسی فعلی能够 علائم و انتقال را کاهش دهند، اما unable به درمان کامل هستند. به این ترتیب، یک نیاز قوی برای روش های جدید و احتمالا درمان کننده برای HSV وجود دارد. در اینجا ما مطالعه قبلی خود را در مورد ویرایش ژن به عنوان یک روش درمانی بالقوه برای HSV در سه جهت مهم گسترش می دهیم. اول، ما یک مدل از реакتیواسیون HSV در موش ها با استفاده از یک مولکول کوچک ایجاد کردیم، تا نشان دهیم که کاهش بار ویروسی گانگلونیک HSV از طریق ویرایش ژن منجر به کاهش قابل توجهی در شیوع ویروس در موش های دارای آلودگی یاوفاسیال می شود. دوم، ما کارایی بالای ویرایش ژن HSV نهفته در DRG پس از آلودگی ژنیتال HSV را نشان دادیم. سوم، ما هپاتوتوکسیتی و نوتوکسیتی را که در برخی از حیوانات تشخیص داده شده بود را با کاهش دوز AAV، ساده سازی رژیم درمانی و استفاده از پروموتورهای خاص سلولی کاهش دادیم یا از بین بردیم. این یافته ها به طور کلی چندین از عوامل اصلی علاقه به درمان HSV را مورد خطاب قرار می دهند و از توسعه بیشتر ویرایش ژن برای آلودگی HSV حمایت می کنند.

موش ها به راحتی به HSV آلوده می شوند و در تعریف بسیاری از جنبه های آلودگی،.latency و کنترل ایمنی HSV نقش مهمی داشته اند. یک محدودیت بزرگ، با این حال، این واقعیت است که آلودگی نهفته HSV در موش ها حداقل تا حدودی یا هیچ واکنش یا شیوع ویروسی در محیط را نشان نمی دهد. بنابراین، موش ها در مطالعه داروهای ضد HSV یا واکسن هایی که برای کنترل آلودگی نهفته و واکنش طراحی شده اند، کاربرد محدودی داشته اند. واکنش HSV در موش ها می تواند با محرک های مختلفی مانند سرکوب ایمنی، استرس هیپرترمیک یا تابش UVB القا شود، اما این روش ها شیوع کم را القا می کنند و می توانند فقط برای سویه های خاص موش یا ایزوله های HSV قابل اجرا باشند. در اینجا ما واکنش HSV را با تزریق داخل صفاقی JQ1، یک مهارکننده دومین برومو که به عنوان یک عامل معکوس کننده latency برای ویروس HIV پیشنهاد شده است، ارزیابی کردیم. قبلا گزارش شده بود که JQ1 able به واکنش HSV در مدل های سلولی latency و القا شیوع از چشم موش های آلوده است، اگرچه مقدار و زمان شیوع به طور کامل مشخص نشده بود. در اینجا ما نشان دادیم که JQ1 able به القا شیوع ویروسی قابل تشخیص در محیط از یک زیرمجموعه قابل توجهی از موش های آلوده، در سطح کمی (10 تا 10 کپی در میلی لیتر) که مشابه آنچه در مطالعات مشاهدتی انسان مشاهده می شود، است.

در بین افراد آلوده به HSV، یک نگرانی اصلی و محرک علاقه به درمان، خطر انتقال ویروس به دیگران است. در حالی که کار قبلی ما کاهش تا 90 درصدی HSV نهفته در گانگل ها پس از ویرایش ژن را نشان داد، هنوز مشخص نبود که چنین کاهش چه تاثیری بر شیوع ویروس در محیط خواهد داشت. با استفاده از مدل واکنش JQ1، ما发现 که کاهش بار ویروسی گانگلونیک از طریق ویرایش ژن影响 قابل توجهی بر شیوع ویروس دارد، هم از نظر تعداد نمونه هایی که ویروس قابل تشخیص دارند و هم از نظر مقدار ویروس دفع شده. در انسان، رابطه بین مقدار شیوع HSV و احتمال انتقال ویروسی هنوز به طور کامل درک نشده است، اما مدل سازی ریاضی قبلی پیشنهاد می کند که کاهش شیوع به سطح زیر 10 کپی ویروسی (مانند آنچه در اکثر حیوانات درمان شده که شیوع باقی مانده بود، مشاهده شد) able به کاهش قابل توجه یا حتی حذف کامل خطر انتقال ویروسی باشد.

یک چالش در مدل واکنش JQ1 در موش ها، طبیعت تصادفی واکنش ویروسی و شیوع القا شده است. در یک گروه داده شده، فقط یک زیرمجموعه از موش های درمان شده با JQ1 شیوع ویروسی قابل تشخیص داشتند و شیوع در یک اپیزود able به پیش بینی شیوع در اپیزودهای بعدی پس از واکنش های مکرر JQ1 نبود. به همین ترتیب، موش های فردی اغلب به طور یک طرفه از یک چشم شیوع داشتند؛ دوباره، این able به پیش بینی طرفیت اپیزودهای بعدی شیوع نبود. این یافته ها مشابه مشاهدات در انسان است، که در آن شیوع اپیزودیک و تصادفی است و می تواند در مکان های مختلف آناتومیک در طول اپیزودهای مختلف شیوع رخ دهد. با این حال، آنها یک چالش در طراحی آزمایش ها برای تشخیص کاهش در شیوع ویروسی ایجاد کردند. در حالی که هر یک از آزمایش های فردی ما یک گرایش مشابه به کاهش شیوع HSV پس از درمان AAV/meganuclease نشان داد، بسیاری از آزمایش های فردی به significancy آماری نرسیدند. این به ویژه در آلودگی ژنیتال HSV مشهود بود، جایی که بار ویروسی نهفته پایین تر در DRG able به نرخ واکنش پایین تر پس از درمان JQ1 بود. بهینه سازی مدل آلودگی ژنیتال برای افزایش بار ویروسی نهفته در DRG ممکن است ارزیابی بیشتر影响 ویرایش ژن بر شیوع ویروسی را تسهیل کند، اگرچه در تجربه ما، تلاش برای افزایش بار ویروسی نهفته باید به دقت با افزایش مرگ و میر حیوانات در فاز حاد آلودگی متعادل شود. در هر صورت، آنالیز متا از داده های ترکیبی از همه آزمایش های ما یک کاهش قابل توجه و آماری در شیوع HSV از حیوانات درمان شده با AAV/meganuclease در مقایسه با کنترل ها تایید کرد. این کاهش در شیوع able به دوز و مدت درمان وابسته بود. مشاهده dernière به ویژه در مورد ترجمه بالینی کار امیدبخش است. در اینجا، موش ها حدود یک ماه پس از تجویز AAV/meganuclease برای بار ویروسی گانگلونیک و شیوع ارزیابی شدند، اما داده های ما پیشنهاد می کند که کارایی ویرایش ژن HSV able به ادامه پس از این نقطه است، که ممکن است منجر به کاهش یا حذف کامل شیوع ویروسی در زمان های بعدی شود. آزمایش ها در حال انجام هستند تا ویرایش ژن HSV را در موش ها به مدت یک سال یا بیشتر ارزیابی کنند و تعیین کنند که آیا کارایی کلی در نهایت به اوج می رسد یا خیر.

یافته دوم مهم در مطالعات ما این است که مگا نوکلئازهای تحویل داده شده توسط AAV می توانند به راحتی وارد نورون ها شده و HSV را در DRG، محل تاخیر HSV در بیماری های تناسلی، ویرایش کنند. ما به طور مداوم کاهش بار ویروسی HSV را در SCG (که گانگلیون های خودکار هستند؛ به طور معمول ~ 90٪ کاهش) در مقایسه با TG (گангلیون های حسی؛ به طور معمول ~ 50-60٪ کاهش) مشاهده کرده ایم. این امر احتمال این را به وجود آورد که DRG (همچنین گانگلیون های حسی) نیز ممکن است نرخ های مشابهی از کاهش ژنوم را نشان دهد. با این حال، ویرایش ژن HSV تاخیری در DRG به طور بسیار کارآمد انجام شد (97٪ کاهش)، که نشان می دهد تفاوت در کارایی ها به نوع گانگلیون (حسی در مقابل خودکار) ذاتی نیست. ما در حال حاضر فرضیه را که کارایی相対ي انتقال AAV در گانگلیون های مختلف به طور اصلی توسط نفوذپذیری نسبی سد خونی-گانگلیونی برای هر نوع گانگلیونی هدایت می شود،支持 می کنیم و مجموعه ای از آزمایش ها را برای حل این vấnله در دست انجام داریم.

یکی از محدودیت های کار ما این است که توالی های هدف HSV-1 مگا نوکلئازهای مورد استفاده در این مطالعه در HSV-2 به خوبی حفاظت نشده اند. بنابراین، آزمایش های آلودگی واژینال را با استفاده از HSV-1 انجام دادیم، تا نشان دهیم که به نظر می رسد هیچ مانعی برای درمان بیماری های تناسلی در مقابل بیماری های اوروفاسیال وجود ندارد؛ یعنی نورون های آلوده به تاخیر در هر دو سایت به راحتی در دسترس وکتورهای AAV و همچنین ویرایش مگا نوکلئازهای ژنوم های HSV تاخیری هستند. ما اشاره می کنیم که درمان موفقیت آمیز آلودگی HSV-1 تناسلی یک نتیجه غیر مهم نیست - بیش از نیمی از موارد جدید زخم های تناسلی در ایالات متحده در حال حاضر ناشی از HSV-1 است. با این حال، ما به طور کامل اهمیت HSV-2 به عنوان یک هدف را درک می کنیم، و کار فعلی در آزمایشگاه ما بر توسعه مگا نوکلئازهای ضد HSV-2 و آزمایش آنها در موش ها و سایر مدل های آلودگی متمرکز است. ما به ویژه به هدف قرار دادن مناطق تکراری HSV-2 علاقه مند هستیم، که نتایج ما در اینجا نشان می دهد که می تواند درمان تک نوکلئازی موثر را امکان پذیر کند، و ما پیشنهاد می کنیم که سایت های تکراری یا تکراری باید برای تلاش های ویرایش ژن هدف قرار گیرند.

وکتورهای AAV به طور کلی ایمن در نظر گرفته می شوند، به ویژه در مقایسه با وکتورهای терапی ژن دیگر. با این حال، در دوزهای بالا، وکتورهای AAV می توانند سمیت کبدی را القا کنند، که最初 به عنوان افزایش ترانس آمیناز ظاهر می شود. در دوزهای AAV بالاتر از آنچه در آزمایش های呈 شده در اینجا استفاده شده است، سمیت کبدی می تواند شدید باشد و منجر به نارسایی کبد در چندین مدل حیوانی شده است,. ما因此 از مشاهده فعالیت ضد HSV قوی در دوزهای AAV که در مدل موشی ما به خوبی تحمل می شد، خرسند بودیم. اخیرا، شواهد هیستولوژیک از آسیب عصبی پس از تجویز AAV در موش ها، خرگوش ها، خوک ها، سگ ها و پریمات های غیر انسانی و در حداقل دو شرکت کننده انسانی در خودکشی گزارش شده است, , ,. مکانیسم ایجاد چنین آسیبی هنوز مشخص نیست؛ از بین فرضیه های فعلی، اشباع ظرفیت تاشد پروتئین عصبی و شناسایی TLR9-مدیATED از RNA وکتور یا ترنس ژن وجود دارد.尽管 شواهد هیستولوژیک از آسیب عصبی، علائم بالینی در حیوانات آزمایشگاهی به ندرت وجود دارد، که عمدتاً شامل اختلالات خفیف در راه رفتن یا تعادل, ,است. چنین علائمی فقط در یک بیمار از بین هزاران شرکت کننده انسانی در آزمایش های درمانی ژن تحویل داده شده توسط AAV گزارش شده است. مطابق با این مطالعات دیگر، ما شواهد خفیفی از آسیب عصبی در موش های آزمایشگاهی مشاهده کردیم، که به صورت تخریب عصبی، نکروز و آکسونوپاتی ظاهر می شد، اما موش های ما هیچ تغییر رفتاری همراه را نشان ندادند. نتایج ما با استفاده از یک پروموتور خاص عصبی برای راندن بیان مگا نوکلئاز به некоторهextent surprisinhg بود، به این معنی که نشان می دهد عدم وجود سمی عصبی که به راحتی با استفاده از یک پروموتور اوبیکیت تشخیص داده می شود. اگرچه این نتایج به خودی خود قطعی نیستند، اما حمایت از در نظر گرفتن یک فرضیه جایگزین را که سمی عصبی گانگلیونی به طور غیر مستقیم از طریق اثرات AAV بر سلول های غیر عصبی، chứ نه خود نورون ها، ایجاد می شود، فراهم می کنند. در هر صورت، داده های ما با ساده سازی رژیم، کاهش دوز و محدودیت بیان ترنس ژن نشان می دهد که می توان راه های میانجی را برای از بین بردن هر دو هپاتوتوکسیسیتی و نوروتوکسیسیتی طراحی کرد. آزمایش های بیشتر در مدل های حیوانی جایگزین آلودگی HSV، مانند خوکچه ها یا حتی پریمات های غیر انسانی، بنابراین مهم هستند، و اگر چنین آزمایش هایی کارایی ضد HSV را با یک پروفایل ایمنی قابل قبول تایید کنند، آزمایش های انسانی ممکن است مورد نیاز باشد. این امر توسط گزارش اخیر استفاده از یک استراتژی ویرایش ژن برای حذف HSV از انسان ها در یک آزمایش بالینی آغاز شده توسط محقق، باز، تک سویه و غیر تصادفی با 3 بیمار با کراتیت استرومال هرپتیک شدید و مقاوم به درمان حمایت می شود. در این مطالعه، هیچ برش غیرهدف یا رویداد عارضه سیستمیک در 18 ماه پیگیری پس از جلوگیری از عود ویروسی تشخیص داده نشد.

در تمام مطالعات، موش های ماده 5 تا 8 هفته ای از نژاد Swiss Webster از شرکت Taconic یا Charles River خریداری شده و در соответствии با دستورالعمل های مرکز سرطان Fred Hutch و راهنمای NIH برای مراقبت و استفاده از حیوانات در تحقیقات نگهداری شدند. روش های آزمایشگاهی انجام شده و توسط کمیته استفاده از حیوانات در تحقیقات (IACUC) مرکز سرطان Fred Hutch تایید شد. اسکان، رژیم غذایی، بستر، غنی سازی و چرخه های روشنایی و تاریکی تحت شرایط حاوی حیوانات در سطح 2 زیستی (ABSL2) اجرا شد.

مجموعه ای محدود از آزمایش های предварین در موش های C57BL/6 (Charles River) انجام شد، و ما نرخ های مشابه کاهش بار ویروسی و ویرایش ژن توسط آزمایش T7 را مشاهده کردیم، که نشان می دهد نتایج ما به طور خاص به نژاد بستگی ندارد.

موش‌ها با تزریق داخل صفاقی کتامین (100 میلی گرم بر کیلوگرم) و زایلازین (12 میلی گرم بر کیلوگرم) بیهوش شدند. موش‌ها در هر دو چشم با پخش 10 PFU از HSV1 syn17+ موجود در 4 میکرولیتر پس از سوزاندن قرنی با استفاده از سوزن 28 گيج آلوده شدند.

موش‌ها با تزریق زیرجلدی 2 میلی گرم از Depo-Provera درمان شدند. پنج تا هفت روز بعد، آنها با تزریق داخل صفاقی کتامین (100 میلی گرم بر کیلوگرم) و زایلازین (12 میلی گرم بر کیلوگرم) بیهوش شدند و با استفاده از پیتت پس از پاک کردن لومن واژینال با یک سواب Calginate، به صورت داخل واژینال با 5 × 10 یا 10 PFU از HSV1 syn17+ موجود در 4 میکرولیتر آلوده شدند.

موش‌هایی که با ایزوفلوران بیهوش شده بودند، دوز نشان داده شده از وکتور AAV را از طریق تزریق داخل پوستی یک طرفه پد مو (WP) یا تزریق داخل وریدی با استفاده از تزریق یک طرفه خلفی چشم (RO؛ موش‌های آلوده به HSV اکتاری) یا تزریق وریدی尾 (TV؛ موش‌های آلوده به HSV اکتاری و واژینال) دریافت کردند. بافت‌ها در زمان مشخص شده جمع‌آوری شدند.

همه روش‌های حیوانی توسط کمیته مراقبت و استفاده از حیوانات مؤسسه سرطان fred هچینسون تأیید شد. این مطالعه به دقت طبق توصیه‌های موجود در راهنمای مراقبت و استفاده از حیوانات آزمایشگاهی مؤسسه ملی بهداشت (“راهنما”) انجام شد.

بازآوری JQ1 با تزریق داخل صفاقی (IP) از (+)-JQ-1 (JQ1، MedChemExpress) با دوز 50 میلی گرم بر کیلوگرم در زمان مشخص شده انجام شد. JQ1 از یک محلول استوک (50 میلی گرم در میلی لیتر در DMSO، Sigma) با استفاده از یک به ده رقیق سازی در یک محلول حامل 10 درصد w/v 2-هیدروکسی پروپیل-β-سیکلودکسترین در PBS (Sigma) تهیه شد. به عنوان جایگزین، بازآوری JQ1 با 2 تزریق IP که با 12 ساعت جدا شده بودند، انجام شد که منجر به تشخیص ریزش ویروس در چشم‌های 6 از 9 (66.6٪) موش‌ها (Supplemental Fig. [5] ) شد. یک مجموعه محدود از آزمایش‌ها بر روی موش‌های C57BL/6 (Charles River) انجام شد و ما نرخ‌ها و سینتیک‌های مشابه ریزش بار ویروسی محیطی را مشاهده کردیم، که نشان می‌دهد نتایج بازآوری JQ1 به طور خاص به سویه موش بستگی ندارد. استرس هیپرتورمیک (HS) به روش توصیف شده در قبل انجام شد.

بافت‌های کبدی در پارافین غوطه‌ور شده، برش خورده و با استفاده از منابع به اشتراک گذاشته شده هیستوپاتولوژی تجربی مرکز سرطان فرد هاتچینسون رنگ آمیزی H&E شدند. ICF توسط یک مشاهده‌گر ناشناس شمارش شده و به عنوان تعداد ICF در هر واحد سطح که با استفاده از Fiji تعیین شده بود، بیان شد.

گانگلیون‌های سه‌تایی در پارافین غوطه‌ور شده، برش خورده و با استفاده از منابع به اشتراک گذاشته شده هیستوپاتولوژی تجربی مرکز سرطان فرد هاتچینسون رنگ آمیزی H&E شدند. تغییرات میکروسکوپی بر اساس شدت توسط یک پاتولوژیست دامپزشکی با استفاده از یک سیستم رتبه‌بندی استاندارد ارزیابی شد که در آن 0 = تغییرات معنی‌دار نبود، 1 = حداقل، 2 = خفیف، 3 = متوسط و 4 = شدید بود.

تحلیل‌های آماری برای هر آزمایش فردی با استفاده از GraphPad Prism نسخه 9.4.1 و R انجام شد. آزمون‌ها دو‌طرفه بودند و مقادیر p کوچکتر از 0.05 معنی‌دار در نظر گرفته شدند. آزمون خاصی که برای هر تحلیل استفاده شد در افسانه مربوطه مشخص شده است.

تحلیل‌های متا بر روی داده‌های ترکیبی از همه آزمایش‌ها (Fig. [1] – [6] ) انجام شد. برای ارزیابی اینکه آیا موش‌های درمان شده با AAV/MN کمتر از حیوانات کنترل ویروس را دفع می‌کنند و اینکه آیا frequency دفع ویروس با مدت زمان درمان مگا نوکلیاز کاهش می‌یابد، از مدل‌های خطی ترکیبی کلی (GLMM) استفاده شد که احتمال دفع ویروس (با دفع ویروس تعریف شده به عنوان AUC > 0) را به عنوان تابعی از مدت زمان درمان و دوز به عنوان متغیرهای پیوسته، در حالی که از آزمایش تنظیم می‌کرد، توصیف کرد.截های تصادفی خاص حیوان در مدل گنجانده شدند تا وابستگی‌های درون‌جانداری بین مشاهدات را به دست آورند. ما همچنین در نظر گرفتیم که یک اصطلاح تعاملی بین دوز و مدت زمان را برای ارزیابی اینکه آیا تغییر در احتمال دفع ویروس در طول زمان تحت تأثیر دوز قرار گرفته است (به‌ویژه اینکه آیا با دوز کاهش می‌یابد یا خیر) شامل شود. ارتباط بین دفع ویروس و متغیرهای همبسته (دوز، مدت زمان درمان) به عنوان نسبت‌های شانس (OR) گزارش شده است. معنی‌داری ارتباط بین احتمال دفع ویروس و دوز، مدت زمان درمان و تعاملات آنها با استفاده از آزمون‌های والد دو‌طرفه ارزیابی شد.

برای مطالعه رابطه بین مقدار دفع ویروس و دوز و مدت زمان درمان، از مدل‌های خطی ترکیبی (LMM) استفاده شد که AUC تبدیل شده به لگاریتم 10 را به عنوان تابعی از دوز و مدت زمان درمان، به عنوان متغیرهای پیوسته، در حالی که از آزمایش تنظیم می‌کرد، توصیف کرد. مدل شامل یک تعامل بین دوز و مدت زمان درمان برای ارزیابی اینکه آیا تغییر در AUC در طول زمان تحت تأثیر دوز قرار گرفته است (به‌ویژه اینکه آیا AUC تبدیل شده به لگاریتم 10 با دوز کاهش می‌یابد یا خیر) بود.截های تصادفی خاص حیوان نیز در مدل گنجانده شدند تا وابستگی‌های درون‌جانداری بین مشاهدات را به دست آورند. خطاهای استاندارد ضرایب رگرسیون با استفاده از یک برآوردگر قوی، نوع ساندویچ، تخمین زده شد. ارتباط بین دفع ویروس و متغیرهای همبسته (دوز، مدت زمان درمان) به عنوان نسبت‌های شانس (OR) گزارش شده است. معنی‌داری ارتباط بین AUC تبدیل شده به لگاریتم 10 و دوز، مدت زمان درمان و تعاملات آنها با استفاده از آزمون‌های والد ارزیابی شد. هر دو مجموعه از تحلیل‌ها همچنین مدل‌هایی را ارزیابی کردند که شامل دوز مربع و ریشه مربع دوز برای انجام تحلیل حساسیت و ارزیابی اینکه آیا رابطه بین احتمال دفع ویروس (یا AUC تبدیل شده به لگاریتم 10) و دوز غیر خطی است یا خیر. به همین ترتیب، مدل‌هایی در نظر گرفته شدند که شامل مدت زمان درمان مربع یا ریشه مربع مدت زمان درمان برای انجام تحلیل حساسیت و ارزیابی اینکه آیا روابط بین احتمال دفع (یا AUC تبدیل شده به لگاریتم 10) و مدت زمان درمان غیر خطی است یا خیر.
